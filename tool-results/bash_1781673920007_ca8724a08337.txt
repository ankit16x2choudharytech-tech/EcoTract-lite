import { NextResponse } from "next/server";
import { db } from "@/lib/db";
import {
  hashPassword,
  makeToken,
  toPublicUser,
  verifyPassword,
} from "@/lib/server-auth";

export async function POST(req: Request) {
  try {
    const body = await req.json();
    const { action } = body as {
      action: "signup" | "login";
      email?: string;
      password?: string;
      name?: string;
    };

    const email = (body.email as string | undefined)?.trim().toLowerCase();
    const password = body.password as string | undefined;

    if (!email || !password) {
      return NextResponse.json(
        { error: "Email and password are required." },
        { status: 400 }
      );
    }

    if (action === "signup") {
      const name = (body.name as string | undefined)?.trim();
      if (!name) {
        return NextResponse.json({ error: "Name is required." }, { status: 400 });
      }
      const existing = await db.user.findUnique({ where: { email } });
      if (existing) {
        return NextResponse.json(
          { error: "An account with this email already exists." },
          { status: 409 }
        );
      }
      const user = await db.user.create({
        data: {
          email,
          name,
          password: hashPassword(password),
        },
      });
      const token = makeToken(user.id, user.email);
      return NextResponse.json({ user: toPublicUser(user), token });
    }

    if (action === "login") {
      const user = await db.user.findUnique({ where: { email } });
      if (!user || !verifyPassword(password, user.password)) {
        return NextResponse.json(
          { error: "Invalid email or password." },
          { status: 401 }
        );
      }
      const token = makeToken(user.id, user.email);
      return NextResponse.json({ user: toPublicUser(user), token });
    }

    return NextResponse.json({ error: "Unknown action." }, { status: 400 });
  } catch (e) {
    console.error("[auth] error", e);
    return NextResponse.json(
      { error: "Something went wrong. Please try again." },
      { status: 500 }
    );
  }
}

function yesterdayStr(): string {
  return daysAgoStr(1);
}

/** Update streak & grant daily-login XP based on lastActiveDate. Returns updated user. */
async function touchDailyLogin(userId: string, lastActiveDate: string | null) {
  const today = todayStr();
  const yesterday = yesterdayStr();

  if (lastActiveDate === today) {
    // already touched today
    return db.user.findUnique({ where: { id: userId } });
  }

  let newStreak = 1;
  if (lastActiveDate === yesterday) {
    newStreak = (await db.user.findUnique({ where: { id: userId } }))?.streak ?? 0;
    newStreak += 1;
  }

  // Bonus XP at streak milestones
  let bonus = XP_RULES.dailyLogin;
  if (newStreak === 7) bonus += XP_RULES.streak7;
  if (newStreak === 30) bonus += XP_RULES.streak30;
  if (newStreak % 7 === 0 && newStreak !== 7) bonus += XP_RULES.weeklyStreak;

  const updated = await db.user.update({
    where: { id: userId },
    data: {
      streak: newStreak,
      lastActiveDate: today,
      xp: { increment: bonus },
    },
  });
  return updated;
}

}

/** PATCH: increment progress; complete & award XP when target reached. */
export async function PATCH(
  req: Request,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const user = await getAuthUser(req);
    if (!user) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }
    const { id } = await params;
    const body = await req.json();
    const { action } = body as { action: "increment" | "complete" | "reset" };

    const goal = await db.goal.findFirst({ where: { id, userId: user.id } });
    if (!goal) {
      return NextResponse.json({ error: "Goal not found." }, { status: 404 });
    }

    if (action === "reset") {
      const updated = await db.goal.update({
        where: { id },
        data: { progress: 0, completed: false },
      });
      return NextResponse.json({ goal: updated });
    }

    if (goal.completed && action !== "complete") {
      return NextResponse.json({ goal, alreadyCompleted: true });
    }

    let newProgress = goal.progress;
    if (action === "increment") newProgress = goal.progress + 1;
    if (action === "complete") newProgress = goal.target;

    const nowCompleted = newProgress >= goal.target;
    let updatedUser = user;
    let xpAwarded = 0;

    if (nowCompleted && !goal.completed) {
      xpAwarded = goal.xp + XP_RULES.goalCompleted;
      updatedUser = await db.user.update({
        where: { id: user.id },
        data: { xp: { increment: xpAwarded } },
      });
    }

    const updated = await db.goal.update({
      where: { id },
      data: { progress: Math.min(newProgress, goal.target), completed: nowCompleted },
    });

    return NextResponse.json({
      goal: updated,
      user: toPublicUser(updatedUser),
      xpAwarded,
    });
  } catch (e) {
    console.error("[goal patch] error", e);
    return NextResponse.json(
      { error: "Failed to update goal." },
      { status: 500 }
    );
  }
}

import { NextResponse } from "next/server";
import { db } from "@/lib/db";
import { getAuthUser } from "@/lib/server-auth";

type SortKey = "xp" | "streak" | "lowest" | "improvement";

export async function GET(req: Request) {
  const user = await getAuthUser(req);
  if (!user) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }
  const url = new URL(req.url);
  const sort = (url.searchParams.get("sort") ?? "xp") as SortKey;

  const sinceDate = new Date();
  sinceDate.setDate(sinceDate.getDate() - 30);
  const since = sinceDate.toISOString().slice(0, 10);

  // previous 30-day window for "improvement"
  const prevEnd = new Date(sinceDate);
  const prevStart = new Date();
  prevStart.setDate(prevStart.getDate() - 60);
  const prevStartStr = prevStart.toISOString().slice(0, 10);
  const prevEndStr = prevEnd.toISOString().slice(0, 10);

  const users = await db.user.findMany({
    select: {
      id: true,
      name: true,
      xp: true,
      streak: true,
      dailyLogs: { select: { date: true, totalCarbon: true } },
    },
  });

  const entries = users.map((u) => {
    const thisMonth = u.dailyLogs.filter((l) => l.date >= since);
    const monthlyCarbon = thisMonth.reduce((s, l) => s + l.totalCarbon, 0);
    const prevMonth = u.dailyLogs.filter(
      (l) => l.date >= prevStartStr && l.date < prevEndStr
    );
    const prevCarbon = prevMonth.reduce((s, l) => s + l.totalCarbon, 0);
    const currAvg = thisMonth.length ? monthlyCarbon / thisMonth.length : 0;
    const prevAvg = prevMonth.length ? prevCarbon / prevMonth.length : 0;
    const improvement = prevAvg > 0 ? Math.round((prevAvg - currAvg) * 10) / 10 : 0;
    return {
      id: u.id,
      name: u.name,
      xp: u.xp,
      streak: u.streak,
      monthlyCarbon: Math.round(monthlyCarbon * 10) / 10,
      improvement,
      isCurrentUser: u.id === user.id,
    };
  });

  let sorted = [...entries];
  switch (sort) {
    case "streak":
      sorted.sort((a, b) => b.streak - a.streak || b.xp - a.xp);
      break;
    case "lowest":
      // lowest emission — but only count users who have logs
      sorted.sort((a, b) => {
        if (a.monthlyCarbon === 0 && b.monthlyCarbon === 0) return b.xp - a.xp;
        if (a.monthlyCarbon === 0) return 1;
        if (b.monthlyCarbon === 0) return -1;
        return a.monthlyCarbon - b.monthlyCarbon;
      });
      break;
    case "improvement":
      sorted.sort((a, b) => b.improvement - a.improvement);
      break;
    default:
      sorted.sort((a, b) => b.xp - a.xp);
  }

  const ranked = sorted.map((e, i) => ({ ...e, rank: i + 1 }));
  return NextResponse.json({ leaderboard: ranked });
}

function parseReport(
  raw: string,
  ctx: {
    weeklyCarbon: number;
    topEmissionSource: string;
    improvementPct: number;
  }
): WeeklyReport {
  try {
    const cleaned = raw.replace(/```json/gi, "").replace(/```/g, "").trim();
    const parsed = JSON.parse(cleaned) as WeeklyReport;
    return {
      weeklyCarbon: Number(parsed.weeklyCarbon ?? ctx.weeklyCarbon),
      topEmissionSource: String(parsed.topEmissionSource ?? ctx.topEmissionSource),

        parsed.biggestImprovement ??

      ),

      ),

    };

  }

function fallbackReport(ctx: {
